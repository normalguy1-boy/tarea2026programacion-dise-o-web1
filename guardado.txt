function actualizarTotal() {
    const suma = carrito.reduce((acc, item) => acc + item.precio, 0)
    totalParrafo.textContent = `Total: $${suma}`
}
 
function agregarAlCarrito(producto) {
    carrito.push(producto)
 
    const fila = document.createElement("div")
 
    const texto = document.createElement("p")
    texto.textContent = `${producto.nombre} — $${producto.precio}`
 
    const btnQuitar = document.createElement("button")
    btnQuitar.textContent = "✕"
    btnQuitar.addEventListener("click", () => {
        const idx = carrito.indexOf(producto)
        if (idx !== -1) carrito.splice(idx, 1)
        contenedorDelCarrito.removeChild(fila)
        actualizarTotal()
    })
 
    fila.append(texto, btnQuitar)
    contenedorDelCarrito.insertBefore(fila, totalParrafo)
    actualizarTotal()
}
